def cloud_coverage(index)
  coverage = weather_report["hourly"]["cloud_cover"][index]
  if coverage > 80
    "Cloudy"
  elsif coverage > 40
    "Partly Cloudy"
  elsif coverage > 10
    "Some Clouds"
  else
    "Clear Skies"
  end
end
